import Link from 'next/link'
import { getPagesUnderRoute } from 'nextra/context'
import { format, parseISO } from 'date-fns'
import GhostLabel from "@/components/GhostLabel";

export function ExtractBlogsByYear() {
  const allBlogs = getPagesUnderRoute('/blogs');
  const blogsByYear = {};

  allBlogs.forEach(blog => {
    if (blog.frontMatter?.date) {
      const year = parseISO(blog.frontMatter.date).getFullYear();
      if (!blogsByYear[year]) {
        blogsByYear[year] = [];
      }
      blogsByYear[year].push(blog);
    }
  });

  return blogsByYear;
}

export function BlogRow({ page }) {
  const formattedDate = page.frontMatter?.date
    ? format(parseISO(page.frontMatter.date), 'MMM dd')
    : null;
  return (
    <Link
      href={page.route}
      style={{ color: 'inherit', textDecoration: 'none' }}
      className="group flex flex-col md:flex-row justify-between items-start md:items-center gap-[4px] md:gap-[12px] w-full opacity-70 hover:opacity-100 transition-opacity duration-200 ease-in-out my-[10px]"
    >
      <div className="flex gap-[5px] items-center">
        <p className="text-xl group-hover:underline">{page.meta?.title || page.frontMatter?.title || page.name}</p>
        <p className="text-xs transform transition-transform duration-200 ease-in-out group-hover:animate-flicker">{'↗'}</p>
      </div>
      <div className="flex gap-[8px] items-center flex-shrink-0">
        {formattedDate && <p className="opacity-50 text-sm">{formattedDate}</p>}
        {formattedDate && page.frontMatter?.readTime && (
          <p className="opacity-50 text-sm">•</p>
        )}
        {page.frontMatter?.readTime && (
          <p className="opacity-50 text-sm">{page.frontMatter.readTime} mins</p>
        )}
      </div>
    </Link>
  );
}

export function BlogHeader() {
  return (
    <div className="max-w-screen-lg mx-auto pt-4 pb-8 mb-16 border-b border-gray-400 border-opacity-20">
      <h1>
        <span className="font-bold leading-tight lg:text-5xl">Blogs</span>
      </h1>
      <p className="italic text-center opacity-60 font-space-grotesk">
        Read my technical blogs for insights on various tech topics and tutorials.
      </p>
    </div>
  )
}

export function BlogIndex() {
  const blogsByYear = ExtractBlogsByYear();
  const years = Object.keys(blogsByYear).sort((a, b) => b - a);

  return (
    <div className="-mt-0">
      {years.map(year => (
        <div key={year} className="relative mt-40">
          <GhostLabel text={year.toString()} />
          {blogsByYear[year].map((page, index) => (
            <BlogRow key={index} page={page} />
          ))}
        </div>
      ))}
    </div>
  );
}

<BlogHeader />
<BlogIndex />
